Bark Scale#

\( \text{bark}=6\sinh^{-1}\left(\frac{f}{600}\right) \)

import numpy as np

def freq_to_bark (freq : float) -> float:
  bark = 6*np.arcsinh(freq/600)
  return bark

avg['F1_bark'] = avg['avg_F1_Hz'].apply(freq_to_bark)
avg['F2_bark'] = avg['avg_F2_Hz'].apply(freq_to_bark)
avg.drop(columns=['avg_F1_Hz','avg_F2_Hz']).to_csv('lab3_bark.csv')
